def highlight_goals(goals):
    # Filter goals for Zero Hunger and Reduced Inequalities
    important_goals = []
    for goal in goals:
        if goal['name'] in ['Zero Hunger', 'Reduced Inequalities']:
            important_goals.append(goal)
    return important_goals

goals = [
    {'name': 'No Poverty', 'index': 1},
    {'name': 'Zero Hunger', 'index': 2},
    {'name': 'Reduced Inequalities', 'index': 10},
]

highlighted = highlight_goals(goals)
print("Highlighted Goals:", highlighted)
